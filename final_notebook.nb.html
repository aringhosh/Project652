# side note function
fun1 <- function(cm){
  accuracy = cm$overall[[1]]
  sensitivity = cm$byClass[[1]]
  specificity = cm$byClass[[2]]
  result <- as.data.frame(c(accuracy, sensitivity, specificity))
  rownames(result) <- c('accuracy', 'sensitivity', 'specificity')
  colnames(result) <- 'values'
  return(result)
}
